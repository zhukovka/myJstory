Regular Expression Features

Named Captures


const re = /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/u;
const result = re.exec('2018-06-09');
// result.groups.year === '2018'
// result.groups.month === '06'
// result.groups.day === '09'

Named Captures


// Named backreferences.
/(?<LowerCaseX>x)y\k<LowerCaseX>/.test('xyx');  // true

// String replacement.
const re = /(?<fst>a)(?<snd>b)/;
'ab'.replace(re, '$<snd>$<fst>');  // 'ba'

Unicode Property Escapes


/\p{Number}/u.test('①');
/\p{Alphabetic}/u.test('雪');
/^\p{Math}+$/u.test('∛∞∉');
/^\p{Script_Extensions=Hiragana}+$/u.test('ひらがな');
